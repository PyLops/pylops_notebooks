nx = 101
x = np.zeros(nx)
x[nx//2] = 1
x[nx//3] = 1
x[nx-4] = 1

Dop = FirstDerivative(nx, edge=True)
Dconvop = Convolve1D(nx, np.array([0.5, 0, -0.5]), offset=1)

y = Dop*x
yconv = Dconvop*x

xadj = Dop.H*y
xadjconv = Dconvop.H*y

print(np.allclose(y, yconv))
print(np.allclose(xadj, xadjconv))

plt.figure()
plt.plot(y, 'k')
plt.plot(yconv, '--r')

plt.figure()
plt.plot(xadj, 'k')
plt.plot(xadjconv, '--r');
