def conjgrad(A, b, x, niter):
    'Conjugate-gradient algorithm for minimizing |A x - dat|^2'
    r = b - A._matvec(x)
    d = r.copy()
    k = r.dot(r)
    for iter in range(niter):
        #if iter == 0:
        #  Ad = A._matvec(r)
        #  dAd = r.dot(Ad)
        #else:
        Ad = A._matvec(d)
        dAd = d.dot(Ad)
        a = k / dAd
        x += a*d
        r -= a*Ad
        kold = k
        k = r.dot(r)
        b = k / kold
        d = r + b*d;
    return x
  
def conjgrad_gpu(A, b, x, niter):
    'Conjugate-gradient algorithm for minimizing |A x - dat|^2'
    r = b - A.matmul(x)
    d = r.clone()
    kold = torch.sum(r*r)
    for iter in range(niter):
        Ad = A.matmul(d)
        dAd = torch.sum(d*Ad)
        a = kold / dAd
        x += a*d
        r -= a*Ad
        k = torch.sum(r*r)
        b = k / kold
        d = r + b*d;
        kold = k
    return x
